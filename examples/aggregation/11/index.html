Query

{
  "group": {
    "into": "g", 
    "by": "p.category"
  }, 
  "as": "p", 
  "from": "Products", 
  "select": {
    "category": "g.key", 
    "cheapestProducts": {
      "where": "p.unitPrice == g.min(p.unitPrice)", 
      "as": "p", 
      "from": "g", 
      "select": "p"
    }
  }
}
