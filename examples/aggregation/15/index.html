jsonq.org

A JSON-based universal data access language

Examples

Query

{
  "group": {
    "into": "g", 
    "by": "p.category"
  }, 
  "as": "p", 
  "from": "Products", 
  "select": {
    "category": "g.key", 
    "mostExpensiveProducts": {
      "where": "p.unitPrice == g.max(p.unitPrice)", 
      "as": "p", 
      "from": "g", 
      "select": "p"
    }
  }
}
