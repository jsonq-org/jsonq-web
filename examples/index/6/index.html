Grouping - nested

Query

{
  "as": "c", 
  "from": "Customers", 
  "select": {
    "yearGroups": {
      "group": {
        "into": "yg", 
        "by": "o.orderDate.year"
      }, 
      "as": "o", 
      "from": "c.Orders", 
      "select": {
        "monthGroups": {
          "group": {
            "into": "mg", 
            "by": "o.orderDate.month"
          }, 
          "as": "o", 
          "from": "yg", 
          "select": {
            "orders": "mg", 
            "month": "mg.key"
          }
        }, 
        "year": "yg.key"
      }
    }, 
    "companyName": "c.companyName"
  }
}

Left-outer join

Arguments

[
  [
    "Beverages", 
    "Condiments", 
    "Vegetables", 
    "Dairy Products", 
    "Seafood"
  ]
]

Query

{
  "join": {
    "into": "ps", 
    "on": "Products", 
    "as": "p", 
    "outer": "left", 
    "by": "c == p.category"
  }, 
  "as": "c", 
  "from": "args[0]", 
  "query": {
    "as": "p", 
    "from": "ps", 
    "select": {
      "category": "c", 
      "productName": "(null == p ? \"(No products)\" : p.productName)"
    }
  }
}
