jsonq.org

A JSON-based universal data access language

Examples

Arguments

[
  [
    "from   ", 
    " salt", 
    " earn ", 
    "  last   ", 
    " near ", 
    " form  "
  ]
]

Query

{
  "group": {
    "using": "anagram_comparator", 
    "into": "words", 
    "by": "w.trim()"
  }, 
  "as": "w", 
  "from": "args[0]", 
  "select": {
    "words": "words.toUpper()", 
    "key": "words.key"
  }
}

Results

[
  {
    "words": [
      "FROM", 
      "FORM"
    ], 
    "key": "fomr"
  }, 
  {
    "words": [
      "SALT", 
      "LAST"
    ], 
    "key": "alst"
  }, 
  {
    "words": [
      "EARN", 
      "NEAR"
    ], 
    "key": "aenr"
  }
]
